($R*, MC$) Assume $\mathbf{D_1}$ ends with an $MC$ on $A'$, which by assumption must be of a lower rank than $A$.
	\begin{prooftree}
		\noLine
		\AXC{$\mathbf{D_0}$}
		\UIC{$\Gamma \Rightarrow A$}

		\noLine
		\AXC{$\mathbf{D_1}'$}
		\UIC{$\Sigma , (\nabla^l A)^n \Rightarrow A'$}

		\noLine
		\AXC{$\mathbf{D_1}''$}
		\UIC{$\Pi , (\nabla^{l'} A')^{n'} \Rightarrow \Delta$}

		\RightLabel{$MC$}
		\BIC{$\nabla^{l'} \Sigma , (\nabla^{l+l'} A)^n , \Pi \Rightarrow \Delta$}

		\dashedLine\RightLabel{$MC$}
		\BIC{$\nabla^{l+l'} \Gamma , \nabla^{l'} \Sigma , \Pi \Rightarrow \Delta$}
	\end{prooftree}
	We can use the induction hypothesis to remove $A$ first, then cut $A'$.
	\begin{prooftree}
		\noLine
		\AXC{$\mathbf{D_0}$}
		\UIC{$\Gamma \Rightarrow A$}

		\noLine
		\AXC{$\mathbf{D_1}'$}
		\UIC{$\Sigma , (\nabla^l A)^n \Rightarrow  A'$}

		\RightLabel{IH}
		\BIC{$\nabla^l \Gamma , \Sigma \Rightarrow A'$}

		\noLine
		\AXC{$\mathbf{D_1}''$}
		\UIC{$\Pi , (\nabla^{l'} A')^{n'} \Rightarrow \Delta$}

		\RightLabel{$MC$}
		\BIC{$\nabla^{l+l'} \Gamma , \nabla^{l'} \Sigma , \Pi \Rightarrow \Delta$}
	\end{prooftree}
