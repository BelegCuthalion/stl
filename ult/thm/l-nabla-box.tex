\begin{lem} \label{lem:l-nabla-box} $\stl(S) \vdash \nabla (\top \rightarrow A) \Rightarrow A$.
\end{lem}
\begin{proof}\quad
	\begin{prooftree}
		\AXC{}
		\RightLabel{$Ta$}
		\UIC{$\Rightarrow \top$}
	
		\AXC{}
		\RightLabel{$Id$}
		\UIC{$A \Rightarrow A$}
	
		\RightLabel{$L \rightarrow$}
		\BIC{$\nabla (\top \rightarrow A) \Rightarrow A$}
	\end{prooftree}
\end{proof}
