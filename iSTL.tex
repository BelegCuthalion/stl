
\section{iSTL$^-$} An iSTL sequent $\Gamma \Rightarrow \Delta$ is a binary relation between $\Gamma$, a multi-set of formulas, and $\Delta$, a sub-singleton of some formula, defined inductively by the following rules.

		\RightLabel{$Id$}
		\UIC{$A \Rightarrow A$}
	\end{prooftree}

		\AXC{}
		\RightLabel{$Ta$}
		\UIC{$\Rightarrow \top$}
	\end{prooftree}
\columnbreak
	\begin{prooftree}
		\AXC{}
		\RightLabel{$Ex$}
		\UIC{$\bot \Rightarrow$}
	\end{prooftree}

\subsection{Cut}
\begin{prooftree}
	\RightLabel{$Cut$}
	\AXC{$\Gamma \Rightarrow A$}
	\AXC{$\Sigma , A \Rightarrow \Delta$}
	\BIC{$\Gamma , \Sigma \Rightarrow \Delta$}
\end{prooftree}
By $\text{iSTL}$ we mean $\text{iSTL}^- + Cut$

\subsection{Lemma}

\subsubsection{}\label{lem:i-nabla-dist-and} iSTL $\vdash \nabla^n (A \land B) \Rightarrow \nabla^n A \land \nabla^n B$

\textit{Proof}:
\begin{prooftree}
	\AXC{}
	\RightLabel{$Id$}
	\UIC{$A \Rightarrow A$}
	\RightLabel{$L\land_1$}
	\UIC{$A \land B \Rightarrow A$}
	\RightLabel{$N$} \doubleLine
	\UIC{$\nabla^n (A \land B) \Rightarrow \nabla^n A$}

	\AXC{}
	\RightLabel{$Id$}
	\UIC{$B \Rightarrow B$}
	\RightLabel{$L\land_2$}
	\UIC{$A \land B \Rightarrow B$}
	\RightLabel{$N$} \doubleLine	
	\UIC{$\nabla^n (A \land B) \Rightarrow \nabla^n B$}
	
	\RightLabel{$R\land$}
	\BIC{$\nabla^n (A \land B) \Rightarrow \nabla^n A \land \nabla^n B$}
\end{prooftree}

\subsubsection{}\label{lem:i-nabla-dist-or} iSTL $\vdash \nabla (A \lor B) \Rightarrow \nabla A \lor \nabla B$

\textit{Proof}:
\begin{prooftree}
	\AXC{}
	\RightLabel{$Id$}
	\UIC{$\nabla A \Rightarrow \nabla A$}
	\RightLabel{$R\lor_1$}
	\UIC{$\nabla A \Rightarrow \nabla A \lor \nabla B$}
	\RightLabel{$Lw$}
	\UIC{$\nabla A , \top \Rightarrow \nabla A \lor \nabla B$}
	\RightLabel{$R\rightarrow$}
	\UIC{$A \Rightarrow \top \rightarrow (\nabla A \lor \nabla B)$}

	\AXC{}
	\RightLabel{$Id$}
	\UIC{$\nabla B \Rightarrow \nabla B$}
	\RightLabel{$R\lor_2$}
	\UIC{$\nabla B \Rightarrow \nabla A \lor \nabla B$}
	\RightLabel{$Lw$}
	\UIC{$\nabla B , \top \Rightarrow \nabla A \lor \nabla B$}
	\RightLabel{$R\rightarrow$}
	\UIC{$B \Rightarrow \top \rightarrow (\nabla A \lor \nabla B)$}

	\RightLabel{$L\lor$}
	\BIC{$A \lor B \Rightarrow \top \rightarrow (\nabla A \lor \nabla B)$}
	\RightLabel{$N$}
	\UIC{$\nabla (A \lor B) \Rightarrow \nabla (\top \rightarrow (\nabla A \lor \nabla B))$}


	\AXC{}
	\RightLabel{$Ta$}
	\UIC{$\Rightarrow \top$}

	\AXC{}
	\RightLabel{$Id$}
	\UIC{$\nabla A \lor \nabla B \Rightarrow \nabla A \lor \nabla B$}

	\RightLabel{$L\rightarrow$}
	\BIC{$\nabla (\top \rightarrow (\nabla A \lor \nabla B)) \Rightarrow \nabla A \lor \nabla B$}



	\RightLabel{$Cut$}
	\BIC{$\nabla (A \lor B) \Rightarrow \nabla A \lor \nabla B$}
\end{prooftree}

\subsubsection{}\label{lem:i-nabla-n-dist-or} iSTL $\vdash \nabla^n (A \lor B) \Rightarrow \nabla^n A \lor \nabla^n B$

\textit{Proof}: Let $\mathbf{D_1}$ be the proof of lemma \ref{lem:i-nabla-dist-or}. For $n > 1$

\begin{prooftree}
	\AXC{}
	\RightLabel{$\mathbf{D_{n-1}}$}
	\UIC{$\nabla^{n-1} (A \lor B) \Rightarrow \nabla^{n-1} A \lor \nabla^{n-1} B$}
	\RightLabel{$N$}
	\UIC{$\nabla^n (A \lor B) \Rightarrow \nabla (\nabla^{n-1} A \lor \nabla^{n-1} B)$}
	
	\AXC{}
	\RightLabel{$\mathbf{D_1}$ \small$\begin{smallmatrix} A:=\nabla^{n-1} A \\ B:=\nabla^{n-1} B \end{smallmatrix}$}
	\UIC{$\nabla (\nabla^{n-1} A \lor \nabla^{n-1} B) \Rightarrow \nabla^n A \lor \nabla^n B$}
	
	\RightLabel{$Cut$} \LeftLabel{$\mathbf{D_n~:~~~~}$}
	\BIC{$\nabla^n (A \lor B) \Rightarrow \nabla^n A \lor \nabla^n B$}
\end{prooftree}

\subsubsection{}\label{lem:i-nabla-bot} iSTL $\vdash \nabla \bot \Rightarrow \bot$

\textit{Proof}:
\begin{prooftree}
	\AXC{}
	\RightLabel{$Ex$}
	\UIC{$\bot \Rightarrow$}
	\RightLabel{$Rw$}
	\UIC{$\bot \Rightarrow \top \rightarrow \bot$}
	\RightLabel{$N$}
	\UIC{$\nabla \bot \Rightarrow \nabla (\top \rightarrow \bot)$}

	\AXC{}
	\RightLabel{$Ta$}
	\UIC{$\Rightarrow \top$}	
	\AXC{}
	\RightLabel{$Id$}
	\UIC{$\bot \Rightarrow \bot$}
	\RightLabel{$L\rightarrow$}
	\BIC{$\nabla (\top \rightarrow \bot) \Rightarrow \bot$}
	
	\RightLabel{$Cut$}
	\BIC{$\nabla \bot \Rightarrow \bot$}
\end{prooftree}

\subsubsection{}\label{lem:i-nabla-n-bot} iSTL $\vdash \nabla^n \bot \Rightarrow \bot$

\textit{Proof}:
\begin{prooftree}
	\AXC{}
	\RightLabel{\ref{lem:i-nabla-bot}}
	\UIC{$\nabla \bot \Rightarrow \bot$}
	\RightLabel{$N^{(n-1)}$} \doubleLine
	\UIC{$\nabla^n \bot \Rightarrow \nabla^{n-1} \bot$}

	\AXC{}
	\RightLabel{\ref{lem:i-nabla-bot}}
	\UIC{$\nabla \bot \Rightarrow \bot$}
	\RightLabel{$N^{(n-2)}$} \doubleLine
	\UIC{$\nabla^{n-1} \bot \Rightarrow \nabla^{n-2} \bot$}

	\RightLabel{$Cut$}
	\BIC{$\nabla^{n} \bot \Rightarrow \nabla^{n-2} \bot$}
	
	\AXC{}
	\RightLabel{\ref{lem:i-nabla-bot}}
	\UIC{$\nabla \bot \Rightarrow \bot$}
	\RightLabel{$N^{(n-3)}$} \doubleLine
	\UIC{$\nabla^{n-2} \bot \Rightarrow \nabla^{n-3} \bot$}

	\RightLabel{$Cut$}
	\BIC{$\nabla^{n} \bot \Rightarrow \nabla^{n-3} \bot$}
	\noLine
	\UIC{$\vdots$}
	\noLine
	\UIC{$\nabla^n \bot \Rightarrow \nabla \bot$}
	
	\AXC{}
	\RightLabel{\ref{lem:i-nabla-bot}}
	\UIC{$\nabla \bot \Rightarrow \bot$}
	
	\RightLabel{$Cut$}
	\BIC{$\nabla^n \bot \Rightarrow \bot$}
\end{prooftree}

\subsubsection{}\label{lem:i-nabla-dist-imp} iSTL $\vdash \nabla^n (A \rightarrow B) \Rightarrow \nabla^n A \rightarrow \nabla^n B$

\textit{Proof}:
\begin{prooftree}
	\AXC{}
	\RightLabel{$Id$}
	\UIC{$A \Rightarrow A$}
	
	\AXC{}
	\RightLabel{$Id$}
	\UIC{$B \Rightarrow B$}
	\RightLabel{$Lw$}
	\UIC{$A , B \Rightarrow B$}
	
	\RightLabel{$L\rightarrow$}
	\BIC{$\nabla (A \rightarrow B) , A \Rightarrow B$}
	\RightLabel{$N$} \doubleLine
	\UIC{$\nabla^{n+1} (A \rightarrow B) , \nabla^n A \Rightarrow \nabla^n B$}
	\RightLabel{$R\rightarrow$}
	\UIC{$\nabla^n (A \rightarrow B) \Rightarrow \nabla^n A \rightarrow \nabla^n B$}
\end{prooftree}

\subsubsection{}\label{lem:modus-ponens} iSTL $\vdash A , \nabla (A \rightarrow B) \Rightarrow B$

\textit{Proof}:
\begin{prooftree}
	\AXC{}
	\RightLabel{$Id$}
	\UIC{$A \Rightarrow A$}

	\AXC{}
	\RightLabel{$Id$}
	\UIC{$B \Rightarrow B$}
	\RightLabel{$Lw$}
	\UIC{$A , B \Rightarrow B$}

	\RightLabel{$L \rightarrow$}
	\BIC{$A , \nabla (A \rightarrow B) \Rightarrow B$}
\end{prooftree}

\subsubsection{}\label{lem:impl-elim} If iSTL $\vdash \Rightarrow A \rightarrow B$ then iSTL $\vdash A \Rightarrow B$.

\textit{Proof}:
\begin{prooftree}
	\AXC{$\Rightarrow A \rightarrow B$}
	\RightLabel{$N$}
	\UIC{$\Rightarrow \nabla (A \rightarrow B)$}

	\AXC{}
	\RightLabel{\ref{lem:modus-ponens}}
	\UIC{$A, \nabla (A \rightarrow B) \Rightarrow B$}
	
	\RightLabel{$Cut$}
	\BIC{$A \Rightarrow B$}
\end{prooftree}

\subsubsection{}\label{lem:conj-context} iSTL $\vdash \{ A_i \}_{i=1}^n \Rightarrow \bigwedge_{i=1}^n A_i$

\textit{Proof}:
\begin{prooftree}
	\AXC{}
	\RightLabel{$Id$}
	\UIC{$A_1 \Rightarrow A_1$}
	\doubleLine \RightLabel{$Lw$}
	\UIC{$\{ A_i \}_{i=0}^n \Rightarrow A_1$}

	\AXC{}
	\RightLabel{$Id$}
	\UIC{$A_2 \Rightarrow A_2$}
	\doubleLine \RightLabel{$Lw$}
	\UIC{$\{ A_i \}_{i=0}^n \Rightarrow A_2$}

	
	\noLine
	\BIC{$\vdots$}
	\RightLabel{$R\land$}
	\UIC{$\{ A_i \}_{i=1}^n \Rightarrow \bigwedge_{i=1}^{n-2} A_i$}
	
	\AXC{}
	\RightLabel{$Id$}
	\UIC{$A_{n-1} \Rightarrow A_{n-1}$}
	\doubleLine \RightLabel{$Lw$}
	\UIC{$\{ A_i \}_{i=1}^n \Rightarrow A_{n-1}$}

	\RightLabel{$R\land$}
	\BIC{$\{ A_i \}_{i=1}^n \Rightarrow \bigwedge_{i=1}^{n-1} A_i$}

	\AXC{}
	\RightLabel{$Id$}
	\UIC{$A_n \Rightarrow A_n$}
	\doubleLine \RightLabel{$Lw$}
	\UIC{$\{ A_i \}_{i=1}^n \Rightarrow A_n$}

	\RightLabel{$R\land$}
	\BIC{$\{ A_i \}_{i=1}^n \Rightarrow \bigwedge_{i=1}^n A_i$}
\end{prooftree}
