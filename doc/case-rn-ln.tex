($R\nabla, *$) This is the only case where $k$ can be more than $0$.

($\star$) In the case that $k = 0$, like previous cases, the last rule of $\mathbf{D_1}$ (and the structure of $A$) is determined by the last rule of $\mathbf{D_0}$. So we have $A = \nabla B$ for some $B$.
		\begin{prooftree}
			\noLine
			\AXC{$\mathbf{D_0}$}
			\UIC{$\p \mathcal{S} \Rightarrow \nabla B$}
			
			\noLine
			\AXC{$\mathbf{D_1}'$}
			\UIC{$\mathcal{T} , \p^{l+1} B , \p^l (\nabla B)^{n-1} \Rightarrow \Delta$}
			\RightLabel{$L \nabla$}
			\UIC{$\mathcal{T} , \p^l (\nabla B)^n \Rightarrow \Delta$}
			
			\RightLabel{$\nabla Cut$} \dashedLine
			\BIC{$\p^{l+1} \mathcal{S} , \mathcal{T} \Rightarrow \Delta$}
		\end{prooftree}
		First, apply the induction hypothesis for $\mathbf{D_0}$ and $\mathbf{D_1}'$ with $A' = A = \nabla B$ and $k' = 0$. Then remove $B$ by a $\nabla Cut$ on $\mathbf{D_0}$ and the resulting sequent.

		\begin{prooftree}
			\noLine
			\AXC{$\mathbf{D_0}$}
			\UIC{$\p \mathcal{S} \Rightarrow \nabla B$}
			
			
			\noLine
			\AXC{$\mathbf{D_0}$}
			\UIC{$\p \mathcal{S} \Rightarrow \nabla B$}
			
			\noLine
			\AXC{$\mathbf{D_1}'$}
			\UIC{$\mathcal{T} , \p^{l+1} B , \p^l (\nabla B)^{n-1} \Rightarrow \Delta$}
			
			\RightLabel{IH}
			\BIC{$\p^{l+1} \mathcal{S} , \mathcal{T} , \p^{l+1} B \Rightarrow \Delta$}
			
			
			\RightLabel{$\nabla Cut$}
			\BIC{$\p^{l+1} \mathcal{S}^2 , \mathcal{T} \Rightarrow \Delta$}
			
			\RightLabel{$Lc$}
			\UIC{$\p^{l+1} \mathcal{S} , \mathcal{T} \Rightarrow \Delta$}
		\end{prooftree}

		($\star \star$) Now, let $k = k' + 1$ for some $k'$. This does not necessarily determine the structure of $A$. Our construction, however, does not depend on the last rule of $\mathbf{D_1}$.
		\begin{prooftree}
			\noLine
			\AXC{$\mathbf{D_0}'$}
			\UIC{$\mathcal{S} \Rightarrow \nabla^{k'} A$}
			\RightLabel{$R \nabla$}
			\UIC{$\p \mathcal{S} \Rightarrow \nabla^{k'+1} A$}
			
			\noLine
			\AXC{$\mathbf{D_1}$}
			\UIC{$\mathcal{T} , \p^{l+k'+1} A^n \Rightarrow \Delta$}
			
			\RightLabel{$\nabla Cut$} \dashedLine
			\BIC{$\p^{l+1} \mathcal{S} , \mathcal{T} \Rightarrow \Delta$}
		\end{prooftree}
		Induction hypothesis on $\mathbf{D_0}'$ and $\mathbf{D_1}$ does exactly the same.
		\begin{prooftree}
			\noLine
			\AXC{$\mathbf{D_0}'$}
			\UIC{$\mathcal{S} \Rightarrow \nabla^{k'} A$}

			\noLine
			\AXC{$\mathbf{D_1}$}
			\UIC{$\mathcal{T} , \p^{l+k'+1} A^n \Rightarrow \Delta$}
			
			\RightLabel{IH}
			\BIC{$\p^{l+1} \mathcal{S} , \mathcal{T} \Rightarrow \Delta$}
		\end{prooftree}
